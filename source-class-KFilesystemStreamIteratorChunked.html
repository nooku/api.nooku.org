 10 /**
 11  * Chunked FileSystem Stream Iterator
 12  *
 13  * Pulls out chunks from an inner stream iterator and yields the chunks as arrays.
 14  *
 15  * @author  Johan Janssens <https://github.com/johanjanssens>
 16  * @package Koowa\Library\Filesystem\Stream\Iterator
 17  */
 18 class KFilesystemStreamIteratorChunked implements OuterIterator, Countable
 19 {
 20     /**
 21      * The inner stream iterator
 22      *
 23      * @var int
 24      */
 25     private $__iterator;
 26 
 27     /**
 28      * Constructor.
 29      *
 30      * @param KFilesystemStreamIterator $iterator  A FilesystemStream object
 31      */
 32     public function __construct(KFilesystemStreamIterator $iterator)
 33     {
 34         $this->__inneriterator = $iterator;
 35     }
 36 
 37     /**
 38      * Seeks to a given chunk position in the stream
 39      *
 40      * @param int $position
 41      * @throws OutOfBoundsException If the position is not seekable.
 42      * @return void
 43      */
 44     public function seek($position)
 45     {
 46         if ($position > $this->count()) {
 47             throw new OutOfBoundsException('Invalid seek position ('.$position.')');
 48         }
 49 
 50         $chunk_size = $this->getChunkSize();
 51         $position   = $chunk_size * $position;
 52 
 53         $this->getInnerIterator()->seek($position);
 54     }
 55 
 56     /**
 57      * Read data from the stream and advance the pointer
 58      *
 59      * @return string
 60      */
 61     public function current()
 62     {
 63         return $this->getInnerIterator()->current();
 64     }
 65 
 66     /**
 67      * Returns the current position of the stream read/write pointer
 68      *
 69      * @return bool|int|mixed
 70      */
 71     public function key()
 72     {
 73         return $this->getInnerIterator()->key() / $this->getChunkSize();
 74     }
 75 
 76     /**
 77      * Move to the next chunk
 78      *
 79      * @return void
 80      */
 81     public function next()
 82     {
 83         $this->getInnerIterator()->next();
 84     }
 85 
 86     /**
 87      * Rewind to the beginning of the stream
 88      *
 89      * @return void
 90      */
 91     public function rewind()
 92     {
 93         $this->getInnerIterator()->rewind();
 94     }
 95 
 96     /**
 97      * Checks if current position is valid
 98      *
 99      * @return bool
100      */
101     public function valid()
102     {
103         return $this->getInnerIterator()->valid();
104     }
105 
106     /**
107      *  Count total amount of chunks
108      *
109      * @return int
110      */
111     public function count()
112     {
113         return $this->getInnerIterator()->getStream()->getSize() / $this->getChunkSize();
114     }
115 
116     /**
117      * Get the chunk size
118      *
119      * @return integer
120      */
121     public function getChunkSize()
122     {
123         return $this->getInnerIterator()->getChunkSize();
124     }
125 
126     /**
127      * Get the stream object
128      *
129      * @return KFilesystemStreamIterator
130      */
131     public function getInnerIterator()
132     {
133         return $this->__inneriterator;
134     }
135 }
