 167     /**
 168      * Opens the stream
 169      *
 170      * @param string $mode The mode used to open the file, as detailed for fopen().
 171      * @throws BadMethodCallException If open is not supported on the stream
 172      * @throws RuntimeException If the stream cannot be opened.
 173      * @return KFilesystemStreamAbstract|false Return a stream object or FALSE on failure.
 174      */
 175     public function open()
 176     {
 177         if($path = $this->getPath())
 178         {
 179             try
 180             {
 181                 $options = $this->getOptions();
 182 
 183                 if(!empty($options))
 184                 {
 185                     $scheme          = $this->getIdentifier()->getName();
 186                     $context         = stream_context_create(array($scheme => $options));
 187                     $this->_resource = @fopen($path, $this->getMode(), false, $context);
 188                 }
 189                 else $this->_resource = @fopen($path, $this->getMode(), false);
 190 
 191             }
 192             catch (Exception $e) {
 193                 $this->_resource = false;
 194             }
 195 
 196             if ($this->_resource === false)
 197             {
 198                 throw new RuntimeException(sprintf(
 199                     'Failed to open stream "%s" with mode "%s"', $path, $this->getMode()
 200                 ));
 201             }
 202 
 203             return true;
 204         }
 205         else throw new BadMethodCallException('The stream "'.self::getName().'" does not support open.');
 206 
 207         return false;
 208     }

 210     /**
 211      * Reads the specified number of bytes from the current position
 212      *
 213      * If the current position is the end-of-file, you must return an empty string.
 214      *
 215      * @param integer|null $length How many bytes of data from the current position should be returned. Defaults to -1
 216      *                            (use the chunk size, default 8192 bytes).
 217      * @throws BadMethodCallException If read is not supported.
 218      * @throws LogicException If read is not allowed.
 219      * @return string If there are less than count bytes available, return as many as are available. If no more data is
 220      *                available, return either FALSE or an empty string.
 221      */
 222     public function read($length = -1)
 223     {
 224         $result = false;
 225 
 226         if($resource = $this->getResource())
 227         {
 228             if (!$this->isReadable()) {
 229                 throw new LogicException('The stream does not allow read.');
 230             } else {
 231                 $result = fread($resource, $length < 0 ? $this->getChunkSize() : $length);
 232             }
 233         }
 234         else throw new BadMethodCallException('The stream "'.self::getName().'" does not support read.');
 235 
 236         return $result;
 237     }

 346     public function copy($stream)

 350             if (!$stream instanceof KFilesystemStreamInterface && !is_resource($stream) && !get_resource_type($stream) == 'stream')
 351             {
 352                 throw new InvalidArgumentException(sprintf(

 359             } else {

 362

 421     /**
 422      * Flush the data from the stream to another stream
 423      *
 424      * If no target stream is being passed and you have cached data that is not yet stored into the underlying storage,
 425      * you should do so now
 426      *
 427      * @param resource|KFilesystemStreamInterface|null $stream The stream resource to flush the data too
 428      * @param int  $length  The total bytes to flush, if -1 the stream will be flushed until eof. The limit should
 429      *                      lie within the total size of the stream.
 430      * @return boolean Should return TRUE if the cached data was successfully stored (or if there was no data to store),
 431      *                 or FALSE if the data could not be stored.
 432      */
 433     public function flush($stream = null, $length = -1)
 434     {
 435         if($this->getResource() && $stream !== NULL)
 436         {
 437             if (!$stream instanceof KFilesystemStreamInterface && !is_resource($stream) && !get_resource_type($stream) == 'stream')
 438             {
 439                 throw new InvalidArgumentException(sprintf(
 440                     'Stream must be on object implementing the KFilesystemStreamInterface or a resource of type "stream".'
 441                 ));
 442             }
 443 
 444             if($stream instanceof KFilesystemStreamInterface) {
 445                 $resource = $stream->getResource();
 446             } else {
 447                 $resource = $stream;
 448             }
 449 
 450             $range = $length < 0 ? $this->getSize() : $length;
 451 
 452             //Send data chunk
 453             while (!$this->eof() && $this->peek() <= $range) {
 454                 $this->copy($resource);
 455             }
 456 
 457             return true;
 458         }
 459 
 460         return false;
 461     }

 647     /**
 648      * Calculate a hash of a Stream
 649      *
 650      * @param string  $algo Hash algorithm (e.g. md5, crc32, etc)
 651      * @param bool    $raw  Whether or not to use raw output
 652      * @return bool|string Returns false on failure or a hash string on success
 653      */
 654     public function getHash($algo = 'sha1', $raw = false)
 655     {
 656         $result = false;
 657 
 658         if ($this->isReadable() && $this->isSeekable())
 659         {
 660             $current = $this->peek();
 661             if ($this->seek(0) && in_array($algo, hash_algos()))
 662             {
 663                 $hash = hash_init($algo);
 664                 hash_update_stream($hash, $this->_resource);
 665                 $result = hash_final($hash, (bool) $raw);
 666                 $this->seek($current);
 667             }
 668         }
 669 
 670         return $result;
 671     }

 798     /**
 799      * Add a filter in FIFO order
 800      *
 801      * @param mixed $filter An object that implements ObjectInterface, ObjectIdentifier object
 802      *                      or valid identifier string
 803      * @param array $config  An optional array of filter config options
 804      * @return  bool   Returns TRUE if the filter was added, FALSE otherwise
 805      */
 806     public function addFilter($filter, $config = array())
 807     {
 808         $result = false;
 809 
 810         if(is_resource($this->_resource))
 811         {
 812             //Handle custom filters
 813             if(!in_array($filter, stream_get_filters()))
 814             {
 815                 //Create the complete identifier if a partial identifier was passed
 816                 if (is_string($filter) && strpos($filter, '.') === false)
 817                 {
 818                     $identifier = $this->getIdentifier()->toArray();
 819                     $identifier['path'] = array('stream', 'filter');
 820                     $identifier['name'] = $filter;
 821 
 822                     $identifier = $this->getIdentifier($identifier);
 823                 }
 824                 else $identifier = $this->getIdentifier($filter);
 825 
 826                 //Make sure the class
 827                 $class = $this->getObject('manager')->getClass($identifier);
 828 
 829                 if(array_key_exists('KFilesystemStreamFilterInterface', class_implements($class)))
 830                 {
 831                     $filter = $class::getName();
 832 
 833                     if (!empty($filter) && !in_array($filter, stream_get_filters())) {
 834                         stream_filter_register($filter, $class);
 835                     }
 836                 }
 837             }
 838 
 839             //If we have a valid filter name create the filter and append it
 840             if(is_string($filter) && !empty($filter))
 841             {
 842                 $mode = 0;
 843                 if($this->isReadable()) {
 844                     $mode = $mode & STREAM_FILTER_READ;
 845                 }
 846 
 847                 if($this->isWritable()) {
 848                     $mode = $mode & STREAM_FILTER_WRITE;
 849                 }

 853                     $this->_filters[$filter] = $filter;
