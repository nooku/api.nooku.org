13  * @author  Johan Janssens <https://github.com/johanjanssens>
14  * @package Koowa\Library\Filter
15  */
16 class KFilterTrim extends KFilterAbstract implements KFilterTraversable
17 {
18     /**
19      * List of characters provided to the trim() function
20      *
21      * If this is null, then trim() is called with no specific character list, and its default behavior will be invoked,
22      * trimming whitespace.
23      *
24      * @var string|null
25      */
26     protected $_charList = null;
27 
28     /**
29      * Constructor
30      *
31      * @param   KObjectConfig $config Configuration options
32      */
33     public function __construct(KObjectConfig $config)
34     {
35         parent::__construct($config);
36 
37         // List of user-defined tags
38         if(isset($config->char_list)) {
39             $this->_charList = $config->char_list;
40         }
41     }
42 
43     /**
44      * Returns the charList option
45      *
46      * @return string|null
47      */
48     public function getCharList()
49     {
50         return $this->_charList;
51     }
52 
53     /**
54      * Sets the charList option
55      *
56      * @param  string|null $charList
57      * @return KFilterTrim
58      */
59     public function setCharList($charList)
60     {
61         $this->_charList = $charList;
62         return $this;
63     }
64 
65     /**
66      * Validate a value
67      *
68      * @param   mixed  $value Value to be validated
69      * @return  bool    True when the variable is valid
70      */
71     public function validate($value)
72     {
73         return (is_string($value));
74     }
75 
76     /**
77      * Sanitize a value
78      *
79      * Returns the variable with characters stripped from the beginning and end
80      *
81      * @param   mixed   $value Value to be sanitized
82      * @return  string
83      */
84     public function sanitize($value)
85     {
86         if (null === $this->_charList) {
87             return trim((string) $value);
88         } else {
89             return trim((string) $value, $this->_charList);
90         }
91     }
