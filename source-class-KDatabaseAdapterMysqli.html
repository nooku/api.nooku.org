111     /**
112      * Connect to the db
113      *
114      * @throws RuntimeException
115      * @return KDatabaseAdapterMysqli
116      */
117      public function connect()
118      {
119         $oldErrorReporting = error_reporting(0);
120 
121         $mysqli = new mysqli(
122             $this->getConfig()->host,
123             $this->getConfig()->username,
124             $this->getConfig()->password,
125             $this->getConfig()->database,
126             $this->getConfig()->port,
127             $this->getConfig()->socket
128         );
129 
130         error_reporting($oldErrorReporting);
131 
132         if (mysqli_connect_errno()) {
133             throw new RuntimeException('Connect failed: (' . mysqli_connect_errno() . ') ' . mysqli_connect_error(), mysqli_connect_errno());
134         }
135 
136         // If supported, request real datatypes from MySQL instead of returning everything as a string.
137         if (defined('MYSQLI_OPT_INT_AND_FLOAT_NATIVE')) {
138             $mysqli->options(MYSQLI_OPT_INT_AND_FLOAT_NATIVE, true);
139         }
140 
141         $this->_connection = $mysqli;
142         $this->_connected  = true;
143 
144         return $this;
145     }

519     /**
520      * Parse the raw column schema information
521      *
522      * @param   object  $info The raw column schema information
523      * @return  KDatabaseSchemaColumn
524      */
525     protected function _parseColumnInfo($info)
526     {
527         list($type, $length, $scope) = $this->_parseColumnType($info->Type);
528 
529         $column           = new KDatabaseSchemaColumn();
530         $column->name     = $info->Field;
531         $column->type     = $type;
532         $column->length   = $length ? $length : null;
533         $column->scope    = $scope ? (int) $scope : null;
534         $column->default  = $info->Default;
535         $column->required = $info->Null != 'YES';
536         $column->primary  = $info->Key == 'PRI';
537         $column->unique   = ($info->Key == 'UNI' || $info->Key == 'PRI');
538         $column->autoinc  = strpos($info->Extra, 'auto_increment') !== false;
539         $column->filter   = $this->_type_map[$type];
540 
541         // Don't keep "size" for integers.
542         if(substr($type, -3) == 'int') {
543             $column->length = null;
544         }
545 
546         // Get the related fields if the column is primary key or part of a unique multi column index.
547         if($indexes = $this->_table_schema[$info->Table]->indexes)
548         {
549             foreach($indexes as $index)
550             {
551                 // We only deal with composite-unique indexes.
552                 if(count($index) > 1 && !$index[1]->Non_unique)
553                 {
554                     $fields = array();
555                     foreach($index as $field) {
556                         $fields[$field->Column_name] = $field->Column_name;
557                     }
558 
559                     if(array_key_exists($column->name, $fields))
560                     {
561                         unset($fields[$column->name]);
562                         $column->related = array_values($fields);
563                         $column->unique = true;
564                         break;
565                     }
566                 }
567             }
568         }
569 
570         return $column;
571     }

573     /**
574      * Given a raw column specification, parse into datatype, length, and decimal scope.
575      *
576      * @param string $spec The column specification; for example, "VARCHAR(255)" or "NUMERIC(10,2)" or "float(6,2)
577      *                     UNSIGNED" or ENUM('yes','no','maybe')
578      * @return array A sequential array of the column type, size, and scope.
579      */
580     protected function _parseColumnType($spec)
581     {
582         $spec    = strtolower($spec);
583         $type    = null;
584         $length  = null;
585         $scope   = null;
586 
587         // find the type first
588         $type = strtok($spec, '( ');
589 
590         // find the parens, if any
591         if (false !== ($pos = strpos($spec, '(')))
592         {
593             // there were parens, so there's at least a length
594             // remove parens to get the size.
595             $length = trim(substr(strtok($spec, ' '), $pos), '()');
596 
597             if($type != 'enum' && $type != 'set')
598             {
599                 // A comma in the size indicates a scope.
600                 $pos = strpos($length, ',');
601                 if ($pos !== false)
602                 {
603                     $scope  = substr($length, $pos + 1);
604                     $length = substr($length, 0, $pos);
605                 }
606 
607             }
608             else $length = explode(',', str_replace("'", "", $length));
609         }
610 
611         return array($type, $length, $scope);
612     }
