 11  * FileSystem Stream Adapter
 12  *
 13  * A generic stream sitting between PHP and streams implementing KFilesystemStreamInterface. Instances of this class
 14  * are created by PHP itself and therefore are unknown to Koowa object manager.
 15  *
 16  * @author  Johan Janssens <https://github.com/johanjanssens>
 17  * @package Koowa\Library\Filesystem\Stream
 18  */
 19 final class KFilesystemStreamAdapter
 20 {
 21     /**
 22      * The stream object
 23      *
 24      * @var KFilesystemStreamInterface
 25      */
 26     private $__stream;
 27 
 28     /**
 29      * The stream context object
 30      *
 31      * @var resource
 32      */
 33     public $context;
 34 
 35     /**
 36      * Create a stream
 37      *
 38      * If trigger_error flag is set exceptions will be catched and E_USER_WARNING raised using trigger_error(). If the
 39      * flag is not set and the stream cannot be created FALSE will be returned instead.
 40      *
 41      * @param string $url    The stream url
 42      * @param string $mode   The type of access required for this stream. (see Table 1 of the fopen() reference);
 43      * @param bool   $trigger_error Raise errors using trigger_error.
 44      * @return KFilesystemStreamInterface|false
 45      */
 46     public function createStream($url, $mode, $trigger_error = false)
 47     {
 48         try
 49         {
 50             $this->__stream = KObjectManager::getInstance()
 51                 ->getObject('filesystem.stream.factory')
 52                 ->createStream($url, $mode, $this->context);
 53         }
 54         catch(Exception $e)
 55         {
 56             //If this flag is set, you are responsible for raising errors using trigger_error() during
 57             //opening of the stream. If this flag is not set, you should not raise any errors.
 58             if ($trigger_error) {
 59                 trigger_error($e->getMessage(), E_USER_WARNING);
 60             }
 61 
 62             return false;
 63         }
 64 
 65         return $this->__stream;
 66     }
 67 
 68     /**
 69      * Get the stream
 70      *
 71      * @return KFilesystemStreamInterface|null
 72      */
 73     public function getStream()
 74     {
 75         return $this->__stream;
 76     }

 78     /**
 79      * Opens file or URL.
 80      *
 81      * This method is called immediately after the stream is initialized (f.e. by fopen() and file_get_contents()).
 82      *
 83      * @param string    $path     Specifies the URL that was passed to the original function.
 84      * @param string    $mode     The mode used to open the file, as detailed for fopen().
 85      * @param int       $options  Stream options (STREAM_USE_PATH | STREAM_REPORT_ERRORS)
 86      * @param string    $opened_path If the path is opened successfully, and STREAM_USE_PATH is set in options,
 87      *                             opened_path should be set to the full path of the file/resource that was actually
 88      *                             opened.
 89      * @return boolean TRUE on success or FALSE on failure.
 90      */
 91     public function stream_open($path, $mode, $options, &$opened_path)
 92     {
 93         if($stream = $this->createStream($path, $mode, $options & STREAM_REPORT_ERRORS))
 94         {
 95             //If the path is opened successfully, and STREAM_USE_PATH is set in options, opened_path
 96             //should be set to the full path of the file/resource that was actually opened.
 97             if ($options & STREAM_USE_PATH) {
 98                 $opened_path = $stream->getPath();
 99             }
100 
101             return true;
102         }
103 
104         return false;
105     }

159     /**
160      * Advisory file locking.
161      *
162      * This method is called in response to flock(), when file_put_contents() (when flags contains LOCK_EX),
163      * stream_set_blocking() and when closing the stream (LOCK_UN).
164      *
165      * @param integer $operation One of the LOCK_* constants
166      * @return boolean TRUE on success or FALSE on failure.
167      */
168     public function stream_lock($operation)
169     {
170         if ($stream = $this->getStream())
171         {
172             if ($operation === LOCK_UN) {
173                 return $stream->unlock();
174             } else {
175                 return $stream->lock($operation);
176             }
177         }

317     /**
318      * Change stream options.
319      *
320      * This method is called to set options on the stream.
321      *
322      * @param integer $option
323      * @param integer $arg1
324      * @param integer $arg2
325      * @return boolean TRUE on success or FALSE on failure. If option is not implemented, FALSE should be returned.
326      */
327     public function stream_set_option($option, $arg1, $arg2)
328     {
329         if ($stream = $this->getStream())
330         {
331             switch($option)
332             {
333                 case STREAM_OPTION_BLOCKING      : $result = $stream->setBlocking($arg1); break;
334                 case STREAM_OPTION_READ_TIMEOUT  : $result = $stream->setTimeout($arg1, $arg2); break;
335                 case STREAM_OPTION_WRITE_BUFFER  : $result = $stream->setBuffer($arg1, $arg2); break;
336                 default : $result = false;
337             }
338 
339             return $result;
340         }
341 
342         return false;
343     }

348      * This method is called in response to all stat() related functions.
349      *
350      * @param string  $path The file path or URL to stat. Note that in the case of a URL, it must be a :// delimited
351      *                      URL. Other URL forms are not supported.
352      * @param integer $flags Holds additional flags set by the streams API.
353      * @return array Should return as many elements as stat() does. Unknown or unavailable values should be set to a
354      *               rational value (usually 0).
355      */
356     public function url_stat($path, $options)
357     {
358         if($stream = $this->createStream($path, 'r', !($options & STREAM_URL_STAT_QUIET))) {
359             $stream->getInfo($options & STREAM_URL_STAT_LINK);
360         }
361 
362         return false;
363     }
