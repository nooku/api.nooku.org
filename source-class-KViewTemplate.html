 11  * Abstract Template View
 12  *
 13  * @author  Johan Janssens <https://github.com/johanjanssens>
 14  * @package Koowa\Library\View
 15  */
 16 abstract class KViewTemplate extends KViewAbstract
 17 {
 18     /**
 19      * Template identifier (com://APP/COMPONENT.template.NAME)
 20      *
 21      * @var string|object
 22      */
 23     protected $_template;
 24 
 25     /**
 26      * Layout name
 27      *
 28      * @var string
 29      */
 30     protected $_layout;
 31 
 32     /**
 33      * Auto assign
 34      *
 35      * @var boolean
 36      */
 37     protected $_auto_fetch;
 38 
 39     /**
 40      * Constructor
 41      *
 42      * @param   KObjectConfig $config Configuration options
 43      */
 44     public function __construct(KObjectConfig $config)
 45     {
 46         parent::__construct($config);
 47 
 48         //Set the auto fetch
 49         $this->_auto_fetch = $config->auto_fetch;
 50 
 51         //Set the layout
 52         $this->setLayout($config->layout);
 53 
 54         //Set the template object
 55         $this->setTemplate($config->template);
 56     }

 58     /**
 59      * Initializes the config for the object
 60      *
 61      * Called from {@link __construct()} as a first step of object instantiation.
 62      *
 63      * @param   KObjectConfig $config Configuration options
 64      * @return  void
 65      */
 66     protected function _initialize(KObjectConfig $config)
 67     {
 68         $config->append(array(
 69             'auto_fetch'         => true,
 70             'layout'             => '',
 71             'template'           => 'default',
 72             'template_filters'   => array('asset', 'decorator'),
 73             'template_functions' => array(
 74                 'route'   => array($this, 'getRoute'),
 75                 'url'     => array($this, 'getUrl'),
 76                 'title'   => array($this, 'getTitle'),
 77                 'content' => array($this, 'getContent'),
 78             ),
 79         ));
 80 
 81         parent::_initialize($config);
 82     }
 83 
 84     /**
 85      * Return the views output
 86      *
 87      * @param KViewContext  $context A view context object
 88      * @return string  The output of the view
 89      */
 90     protected function _actionRender(KViewContext $context)
 91     {
 92         $format = $this->getFormat(); //format cannot be changed through context
 93         $layout = $context->layout;
 94 
 95         if(!parse_url($layout, PHP_URL_SCHEME))
 96         {
 97             //Handle partial layout paths
 98             if (is_string($layout) && strpos($layout, '.') === false)
 99             {
100                 $identifier = $this->getIdentifier()->toArray();
101                 $identifier['name'] = $layout;
102                 unset($identifier['path'][0]);
103 
104                 $layout = (string) $this->getIdentifier($identifier);
105             }
106         }
107 
108         $data = KObjectConfig::unbox($context->data);
109 
110         //Render the template
111         $this->_content = $this->getTemplate()
112             ->loadFile((string) $layout.'.'.$format)
113             ->setParameters($context->parameters)
114             ->render($data);
115 
116         return parent::_actionRender($context);
117     }
118 
119     /**
120      * Fetch the view data
121      *
122      * This function will always fetch the model state. Model data will only be fetched if the auto_fetch property is
123      * set to TRUE.
124      *
125      * @param KViewContext  $context A view context object
126      * @return void
127      */
128     protected function _fetchData(KViewContext $context)
129     {
130         $model = $this->getModel();
131 
132         //Auto-assign the data from the model
133         if($this->_auto_fetch)
134         {
135             //Set the data
136             $name   = $this->getName();
137             $entity = $model->fetch();
138             $context->data->$name = $entity;
139 
140             //Set the parameters
141             if($this->isCollection())
142             {
143                 $context->parameters = $model->getState()->getValues();
144                 $context->parameters->total = $model->count();
145             }
146             else {
147                 $context->parameters = $entity->getProperties();
148                 $context->parameters->total = 1;
149             }
150         }
151         else $context->parameters = $model->getState()->getValues();
152 
153         //Set the layout and view in the parameters.
154         $context->parameters->layout = $context->layout;
155         $context->parameters->view   = $this->getName();
156     }
157 
158     /**
159      * Get the template object attached to the view
160      *
161      * @throws UnexpectedValueException    If the template doesn't implement the TemplateInterface
162      * @return  KTemplateInterface
163      */
164     public function getTemplate()
165     {
166         if (!$this->_template instanceof KTemplateInterface)
167         {
168             //Make sure we have a template identifier
169             if (!($this->_template instanceof KObjectIdentifier)) {
170                 $this->setTemplate($this->_template);
171             }
172 
173             $options = array(
174                 'filters'   => $this->getConfig()->template_filters,
175                 'functions' => $this->getConfig()->template_functions,
176             );
177 
178             $this->_template = $this->getObject($this->_template, $options);
179 
180             if(!$this->_template instanceof KTemplateInterface)
181             {
182                 throw new UnexpectedValueException(
183                     'Template: '.get_class($this->_template).' does not implement KTemplateInterface'
184                 );
185             }
186         }
187 
188         return $this->_template;
189     }
190 
191     /**
192      * Method to set a template object attached to the view
193      *
194      * @param   mixed   $template An object that implements KObjectInterface, an object that implements
195      *                            KObjectIdentifierInterface or valid identifier string
196      * @throws  UnexpectedValueException    If the identifier is not a table identifier
197      * @return  KViewAbstract
198      */
199     public function setTemplate($template)
200     {
201         if (!($template instanceof KTemplateInterface))
202         {
203             if (is_string($template) && strpos($template, '.') === false)
204             {
205                 $identifier = $this->getIdentifier()->toArray();
206                 $identifier['path'] = array('template');
207                 $identifier['name'] = $template;
208 
209                 $identifier = $this->getIdentifier($identifier);
210             }
211             else $identifier = $this->getIdentifier($template);
212 
213             $template = $identifier;
214         }
215 
216         $this->_template = $template;
217 
218         return $this;
219     }
220 
221     /**
222      * Get the layout
223      *
224      * @return string The layout name
225      */
226     public function getLayout()
227     {
228         return empty($this->_layout) ? 'default' : $this->_layout;
229     }
230 
231     /**
232      * Sets the layout name to use
233      *
234      * @param    string  $layout The template name.
235      * @return   $this
236      */
237     public function setLayout($layout)
238     {
239         $this->_layout = $layout;
240         return $this;
241     }

243     /**
244      * Creates a route based on a full or partial query string.
245      *
246      * This function adds the layout information to the route if a layout has been set
247      *
248      * @param string|array $route   The query string used to create the route
249      * @param boolean $fqr          If TRUE create a fully qualified route. Default TRUE.
250      * @param boolean $escape       If TRUE escapes the route for xml compliance. Default TRUE.
251      * @return  KDispatcherRouterRoute The route
252      */
253     public function getRoute($route = '', $fqr = true, $escape = true)
254     {
255         if(is_string($route)) {
256             parse_str(trim($route), $parts);
257         } else {
258             $parts = $route;
259         }
260 
261         //Check to see if there is component information in the route if not add it
262         if (!isset($parts['component'])) {
263             $parts['component'] = $this->getIdentifier()->package;
264         }
265 
266         //Add the view information to the route if it's not set
267         if (!isset($parts['view'])) {
268             $parts['view'] = $this->getName();
269         }
270 
271         if (!isset($parts['layout']) && !empty($this->_layout))
272         {
273             if (($parts['component'] == $this->getIdentifier()->package) && ($parts['view'] == $this->getName())) {
274                 $parts['layout'] = $this->getLayout();
275             }
276         }
277 
278         return parent::getRoute($parts, $fqr, $escape);
279     }
280 
281     /**
282      * Get the view context
283      *
284      * @return  KViewContext
285      */
286     public function getContext()
287     {
288         $context = new KViewContextTemplate();
289         $context->setSubject($this);
290         $context->setData($this->_data);
291         $context->setLayout($this->getLayout());
292 
293         return $context;
294     }
