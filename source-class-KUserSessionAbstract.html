 77     /**
 78      * Constructor
 79      *
 80      * @param KObjectConfig $config  An optional ObjectConfig object with configuration options
 81      */
 82     public function __construct(KObjectConfig $config)
 83     {
 84         parent::__construct($config);
 85 
 86         //Session write and close handlers are called after destructing objects since PHP 5.0.5.
 87         if (version_compare(phpversion(), '5.4.0', '>=')) {
 88             session_register_shutdown();
 89         } else {
 90             register_shutdown_function('session_write_close');
 91         }
 92 
 93         //Only configure the session if it's not active yet
 94         if(!$this->isActive())
 95         {
 96             //Set the session options
 97             $this->setOptions($config->options);
 98 
 99             //Set the session name
100             if (!empty($config->name)) {
101                 $this->setName($config->name);
102             }
103 
104             //Set the session identifier
105             if (!empty($config->id)) {
106                 $this->setId($config->id);
107             }
108 
109             //Set the session handler
110             $this->setHandler($config->handler, KObjectConfig::unbox($config));
111         }
112 
113         //Set the session namespace
114         $this->setNamespace($config->namespace);
115 
116         //Set lifetime time
117         $this->getContainer('metadata')->setLifetime($config->lifetime);
118     }

155     /**
156      * Sets session.* ini variables
157      *
158      * For convenience we omit 'session.' from the beginning of the keys. Explicitly ignores other ini keys.
159      *
160      * @param array $options Session ini directives array(key => value)
161      * @see http://php.net/session.configuration
162      * @throws LogicException    When setting the options of an active session
163      */
164     public function setOptions($options)
165     {
166         if ($this->isActive()) {
167             throw new LogicException('Cannot change the name of an active session');
168         }
169 
170         $valid = array_flip(self::$_valid_options);
171 
172         //Sets session.* ini variables.
173         foreach ($options as $key => $value)
174         {
175             if (isset($valid[$key])) {
176                 ini_set('session.' . $key, $value);
177             }
178         }
179     }

307      * Method to set a session handler object
308      *
309      * @param mixed $handler An object that implements UserSessionHandlerInterface, ObjectIdentifier object
310      *                       or valid identifier string
311      * @param array $config An optional associative array of configuration settings
312      * @return KUserSession
313      */
314     public function setHandler($handler, $config = array())
315     {
316         if (!($handler instanceof KUserSessionHandlerInterface))
317         {
318             if (is_string($handler) && strpos($handler, '.') === false)
319             {
320                 $identifier = $this->getIdentifier()->toArray();
321                 $identifier['path'] = array('session', 'handler');
322                 $identifier['name'] = $handler;
323                 // reset the handler
324                 $handler = $identifier;
325             }
326             
327             $identifier = $this->getIdentifier($handler);
328 
329             //Set the configuration
330             $identifier->getConfig()->append($config);
331 
332             $handler = $identifier;
333         }
334 
335         $this->_handler = $handler;
336         return $this;
337     }

373     /**
374      * Get the session attribute container object
375      *
376      * If the container does not exist a container will be created on the fly.
377      *
378      * @param   mixed $name An object that implements ObjectInterface, ObjectIdentifier object
379      *                      or valid identifier string
380      * @throws UnexpectedValueException    If the identifier is not a session container identifier
381      * @return KUserSessionContainerInterface
382      */
383     public function getContainer($name)
384     {
385         if (!($name instanceof KObjectIdentifier))
386         {
387             //Create the complete identifier if a partial identifier was passed
388             if (is_string($name) && strpos($name, '.') === false)
389             {
390                 $identifier = $this->getIdentifier()->toArray();
391                 $identifier['path'] = array('session', 'container');
392                 $identifier['name'] = $name;
393 
394                 $identifier = $this->getIdentifier($identifier);
395             }
396             else $identifier = $this->getIdentifier($name);
397         }
398         else $identifier = $name;
399 
400         if (!isset($this->_containers[$identifier->name]))
401         {
402             $container = $this->getObject($identifier);
403 
404             if (!($container instanceof KUserSessionContainerInterface))
405             {
406                 throw new UnexpectedValueException(
407                     'Container: '. get_class($container) .' does not implement KUserSessionContainerInterface'
408                 );
409             }
410 
411             //Load the container from the session
412             $namespace = $this->getNamespace();
413             $container->load($_SESSION[$namespace]);
414 
415             $this->_containers[$container->getIdentifier()->name] = $container;
416         }
417         else $container = $this->_containers[$identifier->name];
418 
419         return $container;
420     }

441     /**
442      * Starts the session storage and load the session data into memory
443      *
444      * @see  session_start()
445      * @throws RuntimeException If something goes wrong starting the session.
446      * @return KUserSession
447      */
448     public function start()
449     {
450         if (!$this->isActive())
451         {
452             //Make sure we have a registered session handler
453             if (!$this->getHandler()->isRegistered()) {
454                 $this->getHandler()->register();
455             }
456 
457             session_cache_limiter('none');
458 
459             if (ini_get('session.use_cookies') && headers_sent()) {
460                 throw new RuntimeException('Failed to start the session because headers have already been sent');
461             }
462 
463             if (!session_start()) {
464                 throw new RuntimeException('Session could not be started');
465             }
466 
467             //Re-load the session containers
468             $this->refesh();
469 
470             // Destroy an expired session
471             if ($this->getContainer('metadata')->isExpired()) {
472                 $this->destroy();
473             }
474         }
475 
476         return $this;
477     }

479     /**
480      * Refresh the session data in the memory containers
481      *
482      * This function will load the data from $_SESSION in the various registered containers, based on the container
483      * namespace.
484      *
485      * @return KUserSession
486      */
487     public function refresh()
488     {
489         //Create the namespace if it doesn't exist
490         $namespace = $this->getNamespace();
491 
492         if(!isset($_SESSION[$namespace])) {
493             $_SESSION[$namespace] = array();
494         }
495 
496         //Re-load the session containers
497         foreach($this->_containers as $container)
498         {
499             $namespace = $this->getNamespace();
500             $container->load($_SESSION[$namespace]);
501         }
502 
503         return $this;
504     }

506     /**
507      * Force the session to be saved and closed.
508      *
509      * Session data is usually stored after your script terminated without the need to call UserSession::close(),
510      * but as session data is locked to prevent concurrent writes only one script may operate on a session at any time.
511      *
512      * When using framesets together with sessions you will experience the frames loading one by one due to this locking.
513      * You can reduce the time needed to load all the frames by ending the session as soon as all changes to session

549     /**
550      * Frees all session variables and destroys all data registered to a session
551      *
552      * This method resets the $_SESSION variable and destroys all of the data associated with the current session in its
553      * storage (file or DB). It forces new session to be started after this method is called. It does also unset the
554      * session cookie.
555      *
556      * @see session_destroy()
557      * @return KUserSession
558      */
559     public function destroy()
560     {
561         if ($this->isActive())
562         {
563             // In order to kill the session altogether, like to log the user out, the session id must also be unset. If
564             // a cookie is used to propagate the session id (default behavior) then the session cookie must be deleted.
565             if (ini_get("session.use_cookies") && isset($_COOKIE[$this->getName()]))
566             {
567                 $params = session_get_cookie_params();
568                 setcookie($this->getName(), '', time() - 42000,
569                     $params["path"],
570                     $params["domain"],
571                     $params["secure"],
572                     $params["httponly"]
573                 );
574             }
575 
576             //Clear the session variable
577             $this->clear();
578 
579             //Destroy the session
580             session_destroy();
581         }
582 
583         return $this;
584     }

586     /**
587      * Migrates the current session to a new session id while maintaining all session data
588      *
589      * Note : fork should not clear the session data in memory only delete the session data from persistent storage.
590      *
591      * @param Boolean $destroy  If TRUE, destroy session when forking.
592      * @param integer $lifetime Sets the cookie lifetime for the session cookie. A null value will leave the system
593      *                          settings unchanged, 0 sets the cookie to expire with browser session. Time is in seconds,
594      *                          and is not a Unix timestamp.
595      * @see  session_regenerate_id()
596      * @throws RuntimeException If an error occurs while forking this storage
597      * @return KUserSession
598      */
599     public function fork($destroy = true, $lifetime = null)
600     {
601         if ($this->isActive())
602         {
603             if ($lifetime !== null) {
604                 ini_set('session.cookie_lifetime', $lifetime);
605             }
606 
607             if (!session_regenerate_id($destroy)) {
608                 throw new RuntimeException('Session could not be forked');
609             }
610         }
611 
612         return $this;
613     }
