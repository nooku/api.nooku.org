 10 /**
 11  * Event
 12  *
 13  * You can call the method stopPropagation() to abort the execution of further listeners in your event listener.
 14  *
 15  * @author  Johan Janssens <https://github.com/johanjanssens>
 16  * @package Koowa\Library\Event
 17  */
 18 class KEvent extends KObjectConfig implements KEventInterface
 19 {
 20     /**
 21      * The propagation state of the event
 22      *
 23      * @var boolean
 24      */
 25     protected $_propagate = true;
 26 
 27     /**
 28      * The event name
 29      *
 30      * @var array
 31      */
 32     protected $_name;
 33 
 34     /**
 35      * Target of the event
 36      *
 37      * @var KObjectInterface
 38      */
 39     protected $_target;
 40 
 41     /**
 42      * Constructor.
 43      *
 44      * @param  string              $name       The event name
 45      * @param  array|Traversable   $attributes An associative array or a Traversable object instance
 46      * @param  KObjectInterface    $target     The event target
 47      */
 48     public function __construct($name = '', $attributes = array(), $target = null)
 49     {
 50         parent::__construct($attributes);
 51 
 52         $this->setName($name);
 53         $this->setTarget($target);
 54         $this->setAttributes($attributes);
 55     }
 56 
 57     /**
 58      * Get the event name
 59      *
 60      * @return string   The event name
 61      */
 62     public function getName()
 63     {
 64         return $this->_name;
 65     }
 66 
 67     /**
 68      * Set the event name
 69      *
 70      * @param string $name  The event name
 71      * @return KEvent
 72      */
 73     public function setName($name)
 74     {
 75         $this->_name = $name;
 76         return $this;
 77     }
 78 
 79     /**
 80      * Get the event target
 81      *
 82      * @return object   The event target
 83      */
 84     public function getTarget()
 85     {
 86         return $this->_target;
 87     }
 88 
 89     /**
 90      * Set the event target
 91      *
 92      * @param mixed $target The event target
 93      * @return KEvent
 94      */
 95     public function setTarget($target)
 96     {
 97         $this->_target = $target;
 98         return $this;
 99     }
100 
101     /**
102      * Set attributes
103      *
104      * Overwrites existing attributes
105      *
106      * @param  array|Traversable $attributes
107      * @throws InvalidArgumentException If the attributes are not an array or are not traversable.
108      * @return KEvent
109      */
110     public function setAttributes($attributes)
111     {
112         if (!is_array($attributes) && !$attributes instanceof Traversable)
113         {
114             throw new InvalidArgumentException(sprintf(
115                 'Event attributes must be an array or an object implementing the Traversable interface; received "%s"', gettype($attributes)
116             ));
117         }
118 
119         //Set the arguments.
120         foreach ($attributes as $key => $value) {
121             $this->set($key, $value);
122         }
123 
124         return $this;
125     }
126 
127     /**
128      * Get all arguments
129      *
130      * @return array
131      */
132     public function getAttributes()
133     {
134         return $this->toArray();
135     }
136 
137     /**
138      * Get an attribute
139      *
140      * If the attribute does not exist, the $default value will be returned.
141      *
142      * @param  string $name The attribute name
143      * @param  mixed $default
144      * @return mixed
145      */
146     public function getAttribute($name, $default = null)
147     {
148        return $this->get($name, $default);
149     }
150 
151     /**
152      * Set an attribute
153      *
154      * @param  string $name The attribute
155      * @param  mixed $value
156      * @return KEvent
157      */
158     public function setAttribute($name, $value)
159     {
160         $this->set($name, $value);
161         return $this;
162     }
163 
164     /**
165      * Returns whether further event listeners should be triggered.
166      *
167      * @return boolean  TRUE if the event can propagate. Otherwise FALSE
168      */
169     public function canPropagate()
170     {
171         return $this->_propagate;
172     }
173 
174     /**
175      * Stops the propagation of the event to further event listeners.
176      *
177      * If multiple event listeners are connected to the same event, no further event listener will be triggered once
178      * any trigger calls stopPropagation().
179      *
180      * @return KEvent
181      */
182     public function stopPropagation()
183     {
184         $this->_propagate = false;
185         return $this;
186     }
187 
188     /**
189      * Get a new instance
190      *
191      * @return KObjectConfig
192      */
193     final public function getInstance()
194     {
195         $instance = new KObjectConfig(array(), $this->_readonly);
196         return $instance;
197     }
198 
199     /**
200      * Get an event property or attribute
201      *
202      * If an event property exists the property will be returned, otherwise the attribute will be returned. If no
203      * property or attribute can be found the method will return NULL.
204      *
205      * @param  string $name    The property name
206      * @param  mixed  $default The default value
207      * @return mixed|null  The property value
208      */
209     final public function get($name, $default = null)
210     {
211         $method = 'get'.ucfirst($name);
212         if(!method_exists($this, $method) ) {
213             $value = parent::get($name);
214         } else {
215             $value = $this->$method();
216         }
217 
218         return $value;
219     }
220 
221     /**
222      * Set an event property or attribute
223      *
224      * If an event property exists the property will be set, otherwise an attribute will be added.
225      *
226      * @param  string $name
227      * @param  mixed  $value
228      * @return KEvent
229      */
230     final public function set($name, $value)
231     {
232         $method = 'set'.ucfirst($name);
233         if(!method_exists($this, $method) ) {
234            parent::set($name, $value);
235         } else {
236             $this->$method($value);
237         }
238 
239         return $this;
240     }
