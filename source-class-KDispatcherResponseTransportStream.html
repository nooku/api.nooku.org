 10 /**
 11  * Stream Dispatcher Response Transport
 12  *
 13  * Streaming is a data transfer mechanism in version HTTP 1.1 in which a web server serves content in a series of chunks.
 14  * Two mechanisms exist to do this : range serving and chunk serving.
 15  *
 16  * -- Range Serving
 17  *
 18  * This mechanism is the process of sending only a portion of the data from a server to a client. Range-serving uses
 19  * the Range HTTP request header and the Accept-Ranges and Content-Range HTTP response headers.
 20  *
 21  * Clients which request range-serving might do so in cases in which a large file has been only partially delivered and
 22  * a limited portion of the file is needed in a particular range. Range Serving is therefore a method of bandwidth
 23  * optimization
 24  *
 25  * -- Chunk Serving
 26  *
 27  * This mechanism uses the Transfer-Encoding HTTP response header instead of the Content-Length header, which the protocol
 28  * would otherwise require. Because the Content-Length header is not used, the server does not need to know the length of
 29  * the content before it starts transmitting a response to the client.
 30  *
 31  * Web servers can begin transmitting responses with dynamically-generated content before knowing the total size of
 32  * that content. The size of each chunk is sent right before the chunk itself so that a client can tell when it has
 33  * finished receiving data for that chunk. The data transfer is terminated by a final chunk of length zero.
 34  *
 35  * @author  Johan Janssens <https://github.com/johanjanssens>
 36  * @package Koowa\Library\Dispatcher\Response\Transport
 37  * @see http://en.wikipedia.org/wiki/Byte_serving
 38  * @see http://en.wikipedia.org/wiki/Chunked_transfer_encoding
 39  * @see http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.16
 40  */
 41 class KDispatcherResponseTransportStream extends KDispatcherResponseTransportHttp
 42 {
 43     /**
 44      * Byte offset
 45      *
 46      * @var integer
 47      */
 48     protected $_offset;
 49 
 50     /**
 51      * Byte range
 52      *
 53      * @var integer
 54      */
 55     protected $_range;
 56 
 57     /**
 58      * Initializes the config for the object
 59      *
 60      * Called from {@link __construct()} as a first step of object instantiation.
 61      *
 62      * @param   KObjectConfig $config  An optional ObjectConfig object with configuration options
 63      * @return  void
 64      */
 65     protected function _initialize(KObjectConfig $config)
 66     {
 67         $config->append(array(
 68             'priority'   => self::PRIORITY_HIGH,
 69         ));
 70 
 71         parent::_initialize($config);
 72     }

 74     /**
 75      * Get the byte offset
 76      *
 77      * @param KDispatcherResponseInterface $response
 78      * @return int The byte offset
 79      * @throws KHttpExceptionRangeNotSatisfied   If the byte offset is outside of the total size of the file
 80      */
 81     public function getOffset(KDispatcherResponseInterface $response)
 82     {
 83         if(!isset($this->_offset))
 84         {
 85             $offset = 0;
 86 
 87             if($response->getRequest()->isStreaming())
 88             {
 89                 $ranges = $response->getRequest()->getRanges();
 90 
 91                 if (!empty($ranges[0]['first'])) {
 92                     $offset = (int) $ranges[0]['first'];
 93                 }
 94 
 95                 if ($offset > $this->getFileSize($response)) {
 96                     throw new KHttpExceptionRangeNotSatisfied('Invalid range');
 97                 }
 98             }
 99 
100             $this->_offset = $offset;
101         }
102 
103         return $this->_offset;
104     }
105 
106     /**
107      * Get the byte range
108      *
109      * @param KDispatcherResponseInterface $response
110      * @return int The last byte offset
111      */
112     public function getRange(KDispatcherResponseInterface $response)
113     {
114         if(!isset($this->_range))
115         {
116             $length = $this->getFileSize($response);
117             $range  = $length - 1;
118 
119             if($response->getRequest()->isStreaming())
120             {
121                 $ranges = $response->getRequest()->getRanges();
122 
123                 if (!empty($ranges[0]['last'])) {
124                     $range = (int) $ranges[0]['last'];
125                 }
126 
127                 if($range > $length - 1) {
128                     $range = $length - 1;
129                 }
130             }
131 
132             $this->_range = $range;
133         }
134 
135         return $this->_range;
136     }
137 
138     /**
139      * Get the file size
140      *
141      * @param KDispatcherResponseInterface $response
142      * @return int The file size in bytes
143      */
144     public function getFileSize(KDispatcherResponseInterface $response)
145     {
146         return $response->getStream()->getSize();
147     }

149     /**
150      * Generate an etag from a file stream
151      *
152      * This functions returns a md5 hash of same format as Apache does. Eg "%ino-%size-%mtime" using the file info.
153      * @link http://stackoverflow.com/questions/44937/how-do-you-make-an-etag-that-matches-apache
154      *
155      * @param KDispatcherResponseInterface $response
156      * @return string
157      */
158     public function getFileEtag(KDispatcherResponseInterface $response)
159     {
160         $info = $response->getStream()->getInfo();
161         $etag = sprintf('"%x-%x-%s"', $info['ino'], $info['size'],base_convert(str_pad($info['mtime'],16,"0"),10,16));
162 
163         return $etag;
164     }

166     /**
167      * Sends content for the current web response.
168      *
169      * We flush(stream) the data to the output buffer based on the chunk size and range information provided in the
170      * request. The default chunk size is 8 MB.
171      *
172      * @param KDispatcherResponseInterface $response
173      * @return KDispatcherResponseTransportRedirect
174      */
175     public function sendContent(KDispatcherResponseInterface $response)
176     {
177         if ($response->isSuccess())
178         {
179             //For a certain unmentionable browser
180             if(ini_get('zlib.output_compression')) {
181                 @ini_set('zlib.output_compression', 'Off');
182             }
183 
184             //Fix for IE7/8
185             if(function_exists('apache_setenv')) {
186                 @apache_setenv('no-gzip', '1');
187             }
188 
189             //Remove PHP time limit
190             if(!ini_get('safe_mode')) {
191                 @set_time_limit(0);
192             }
193 
194             //Make sure the output buffers are cleared
195             $level = ob_get_level();
196             while($level > 0) {
197                 ob_end_clean();
198                 $level--;
199             }
200 
201             $stream  = $response->getStream();
202 
203             $offset = $this->getOffset($response);
204             $range  = $this->getRange($response);
205 
206             if ($offset > 0) {
207                 $stream->seek($offset);
208             }
209 
210             $output = fopen('php://output', 'w+');
211             $stream->flush($output, $range);
212             $stream->close();
213             fclose($output);
214 
215             return $this;
216         }
217 
218         parent::sendContent($response);
219     }
220 
221     /**
222      * Send HTTP response
223      *
224      * @param KDispatcherResponseInterface $response
225      * @return boolean
226      */
227     public function send(KDispatcherResponseInterface $response)
228     {
229         $request  = $response->getRequest();
230 
231         if($response->isStreamable())
232         {
233             //Explicitly set the Accept Ranges header to bytes to inform client we accept range requests
234             $response->headers->set('Accept-Ranges', 'bytes');
235 
236             //Set a file etag
237             $response->headers->set('etag', $this->getFileEtag($response));
238 
239             if($request->isStreaming())
240             {
241                 if($response->isSuccess())
242                 {
243                     //Default Content-Type Header
244                     if(!$response->headers->has('Content-Type')) {
245                         $response->headers->set('Content-Type', 'application/octet-stream');
246                     }
247 
248                     //Content Range Headers
249                     $offset = $this->getOffset($response);
250                     $range  = $this->getRange($response);
251                     $size   = $this->getFileSize($response);
252 
253                     $response->setStatus(KHttpResponse::PARTIAL_CONTENT);
254                     $response->headers->set('Content-Length', $range - $offset + 1);
255                     $response->headers->set('Content-Range', sprintf('bytes %s-%s/%s', $offset, $range, $size));
256                 }
257 
258                 if($response->isError())
259                 {
260                     /**
261                      * A server sending a response with status code 416 (Requested range not satisfiable) SHOULD include a
262                      * Content-Range field with a byte-range- resp-spec of "*". The instance-length specifies the current
263                      * length of the selected resource.
264                      *
265                      * @see : http://www.w3.org/Protocols/rfc2616/rfc2616-sec14.html#sec14.16
266                      */
267                     if($response->getStatusCode() == KHttpResponse::REQUESTED_RANGE_NOT_SATISFIED)
268                     {
269                         $size = $this->getFileSize($response);
270                         $response->headers->set('Content-Range', sprintf('bytes */%s', $size));
271                     }
272                 }
273             }
274         }
275 
276         return parent::send($response);
277     }
