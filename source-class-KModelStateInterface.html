 10 /**
 11  * Model State Interface
 12  *
 13  * @author  Johan Janssens <https://github.com/johanjanssens>
 14  * @package Koowa\Library\Model\State
 15  */
 16 interface KModelStateInterface
 17 {
 18     /**
 19      * Insert a new state
 20      *
 21      * @param   string   $name     The name of the state
 22      * @param   mixed    $filter   Filter(s), can be a FilterInterface object, a filter name or an array of filter names
 23      * @param   mixed    $default  The default value of the state
 24      * @param   boolean  $unique   TRUE if the state uniquely identifies an entity, FALSE otherwise. Default FALSE.
 25      * @param   array    $required Array of required states to determine if the state is unique. Only applicable if the state is unqiue.
 26      * @param   boolean  $internal If TRUE the state will be considered internal and should not be included in a routes.
 27      *                             Default FALSE.
 28      * @return  KModelStateInterface
 29      */
 30     public function insert($name, $filter, $default = null, $unique = false, $required = array(), $internal = false);
 31 
 32     /**
 33      * Retrieve a configuration item and return $default if there is no element set.
 34      *
 35      * @param string $name      The state name
 36      * @param mixed  $default   The state default value if no state can be found
 37      * @return mixed
 38      */
 39     public function get($name, $default = null);
 40 
 41     /**
 42      * Set state value
 43      *
 44      * @param   string  $name The state name.
 45      * @param   mixed   $value The state value.
 46      * @return  KModelStateInterface
 47      */
 48     public function set($name, $value = null);
 49 
 50     /**
 51      * Check if a state exists
 52      *
 53      * @param   string  $name The state name.
 54      * @return  boolean
 55      */
 56     public function has($name);
 57 
 58     /**
 59      * Remove an existing state
 60      *
 61      * @param   string $name The name of the state
 62      * @return  KModelStateInterface
 63      */
 64     public function remove($name);
 65 
 66     /**
 67      * Reset all state data and revert to the default state
 68      *
 69      * @return KModelStateInterface
 70      */
 71     public function reset();
 72 
 73     /**
 74      * Get the total number of entities
 75      *
 76      * @return  int
 77      */
 78     public function count();
 79 
 80     /**
 81      * Set the state data
 82      *
 83      * This function will only filter values if we have a value. If the value is an empty string it will be filtered
 84      * to NULL.
 85      *
 86      * @param   array $data An associative array of state values by name
 87      * @return  KModelState
 88      */
 89     public function setValues(array $data);
 90 
 91     /**
 92      * Get the state data
 93      *
 94      * This function only returns states that have been been set.
 95      *
 96      * @param   boolean $unique If TRUE only retrieve unique state values, default FALSE
 97      * @return  array   An associative array of state values by name
 98      */
 99     public function getValues($unique = false);
100 
101     /**
102      * Set a state property
103      *
104      * @param string $name      The name of the state
105      * @param string $property  The name of the property
106      * @param mixed  $value     The value of the property
107      * @return KModelStateInterface
108      */
109     public function setProperty($name, $property, $value);
110 
111     /**
112      * Get a state property
113      *
114      * @param string $name     The name of the state
115      * @param string $property The name of the property
116      * @return mixed
117      */
118     public function getProperty($name, $property);
119 
120     /**
121      * Check if a state property exists
122      *
123      * @param string $name     The name of the state
124      * @param string $property The name of the property
125      * @return boolean   Return TRUE if the the property exists, FALSE otherwise
126      */
127     public function hasProperty($name, $property);
128 
129     /**
130      * Check if the state information is unique
131      *
132      * @return  boolean TRUE if the state is unique, otherwise FALSE.
133      */
134     public function isUnique();
135 
136     /**
137      * Check if the state information is empty
138      *
139      * @param   array   $exclude An array of states names to exclude.
140      * @return  boolean TRUE if the state is empty, otherwise FALSE.
141      */
142     public function isEmpty(array $exclude = array());
143 }
