 32     /**
 33      * Constructor
 34      *
 35      * @param   KObjectConfig $config Configuration options
 36      */
 37     public function __construct(KObjectConfig $config)
 38     {
 39         parent::__construct($config);
 40 
 41         $this->addCommandCallback('before.read' , 'setReferrer');
 42         $this->addCommandCallback('after.apply' , '_lockReferrer');
 43         $this->addCommandCallback('after.read'  , '_unlockReferrer');
 44         $this->addCommandCallback('after.save'  , '_unsetReferrer');
 45         $this->addCommandCallback('after.cancel', '_unsetReferrer');
 46 
 47         $this->addCommandCallback('after.read'  , '_lockResource');
 48         $this->addCommandCallback('after.save'  , '_unlockResource');
 49         $this->addCommandCallback('after.cancel', '_unlockResource');
 50 
 51         $this->_cookie_path = $config->cookie_path;
 52         $this->_cookie_name = $config->cookie_name;
 53     }

 55     /**
 56      * Initializes the options for the object
 57      *
 58      * Called from {@link __construct()} as a first step of object instantiation.
 59      *
 60      * @param  KObjectConfig $config A ObjectConfig object with configuration options
 61      * @return void
 62      */
 63     protected function _initialize(KObjectConfig $config)
 64     {
 65         $config->append(array(
 66             'cookie_name' => 'referrer',
 67             'cookie_path' => $this->getObject('request')->getBaseUrl()->toString(KHttpUrl::PATH)
 68         ));
 69 
 70         parent::_initialize($config);

 74      * Check if the behavior is supported
 75      *
 76      * @return  boolean  True on success, false otherwise
 77      */
 78     public function isSupported()
 79     {
 80         $mixer   = $this->getMixer();
 81         $request = $mixer->getRequest();
 82 
 83         if ($mixer instanceof KControllerModellable && $mixer->isDispatched() && $request->getFormat() == 'html') {

 90     /**
 91      * Get the referrer
 92      *
 93      * @param   KControllerContextInterface $context A controller context object
 94      * @return  KHttpUrl    A HttpUrl object
 95      */
 96     public function getReferrer(KControllerContextInterface $context)
 97     {
 98         if($context->request->cookies->has($this->_cookie_name))
 99         {
100             $referrer = $context->request->cookies->get($this->_cookie_name, 'url');
101             $referrer = $this->getObject('lib:http.url', array('url' => $referrer));
102         }
103         else $referrer = $this->findReferrer($context);
104 
105         return $referrer;
106     }

108     /**
109      * Set the referrer
110      *
111      * @param  KControllerContextInterface $context A controller context object
112      * @return void
113      */
114     public function setReferrer(KControllerContextInterface $context)
115     {
116         if (!$context->request->cookies->has($this->_cookie_name.'_locked'))
117         {
118             $request  = $context->request->getUrl();
119             $referrer = $context->request->getReferrer();
120 
121             //Compare request url and referrer
122             if (isset($referrer) && !$request->equals($referrer))
123             {
124                 //Add the referrer cookie
125                 $cookie = $this->getObject('lib:http.cookie', array(
126                     'name'   => $this->_cookie_name,
127                     'value'  => $referrer,
128                     'path'   => $this->_cookie_path
129                 ));
130 
131                 $context->response->headers->addCookie($cookie);
132             }
133         }
134     }

136     /**
137      * Find the referrer based on the context
138      *
139      * Method is being called when no referrer can be found in the request or when request url and referrer are
140      * identical. Function should return a url that is different from the request url to avoid redirect loops.
141      *
142      * @param KControllerContextInterface $context
143      * @return KHttpUrl    A HttpUrl object
144      */
145     public function findReferrer(KControllerContextInterface $context)
146     {
147         $controller = $this->getMixer();
148         $identifier = $controller->getIdentifier();
149 
150         $component = $identifier->package;
151         $view      = KStringInflector::pluralize($identifier->name);
152         $referrer  = $controller->getView()->getRoute('component=' . $component . '&view=' . $view, true, false);
153 
154         return $this->getObject('lib:http.url', array('url' => $referrer));
155     }

200      *

203     public function isLocked()
204     {
205         if($this->getModel()->getState()->isUnique())
206         {
207             $entity = $this->getModel()->fetch();
208 
209             if($entity->isLockable() && $entity->isLocked()) {
210                 return true;

217     /**
218      * Check if the resource is lockable
219      *
220      * @return bool Returns TRUE if the resource is can be locked, FALSE otherwise.
221      */
222     public function isLockable()
223     {
224         $controller = $this->getMixer();
225 
226         if($controller instanceof KControllerModellable)
227         {
228             if($this->getModel()->getState()->isUnique())
229             {
230                 $entity = $this->getModel()->fetch();
231 
232                 if($entity->isLockable()) {
233                     return true;
234                 }
235             }
236         }
237 
238         return false;

279         if($this->getRequest()->getFormat() == 'html')
280         {

326     /**
327      * Save action
328      *
329      * This function wraps around the edit or add action. If the model state is unique a edit action will be
330      * executed, if not unique an add action will be executed.
331      *
332      * This function also sets the redirect to the referrer if the action succeeds and will redirect to the
333      * current url if the edit/add action fails while setting the status message.
334      *
335      * @param   KControllerContextInterface  $context A controller context object
336      * @return  KModelEntityInterface
337      */
338     protected function _actionSave(KControllerContextInterface $context)
339     {
340         $action = $this->getModel()->getState()->isUnique() ? 'edit' : 'add';
341         $entity = $context->getSubject()->execute($action, $context);
342 
343         //Create the redirect
344         if($entity->getStatus() === KModelEntityInterface::STATUS_FAILED)
345         {
346             $url     = $context->request->getReferrer();
347             $message = $entity->getStatusMessage() ? $entity->getStatusMessage() : ucfirst($action).' Action Failed';
348 
349             $context->response->setRedirect($url, $message, KControllerResponseInterface::FLASH_ERROR);
350         }
351         else  $context->response->setRedirect($this->getReferrer($context));
352 
353         return $entity;
354     }
355 
356     /**
357      * Apply action
358      *
359      * This function wraps around the edit or add action. If the model state is unique a edit action will be
360      * executed, if not unique an add action will be executed.
361      *
362      * This function also sets the redirect to the current url for 'add' actions and will redirect to current
363      * url if the edit/add action fails while setting the status message.
364      *
365      * @param    KControllerContextInterface  $context A controller context object
366      * @return   KModelEntityInterface
367      */
368     protected function _actionApply(KControllerContextInterface $context)
369     {
370         $action = $this->getModel()->getState()->isUnique() ? 'edit' : 'add';
371         $entity = $context->getSubject()->execute($action, $context);
372 
373         if($entity->getStatus() !== KModelEntityInterface::STATUS_FAILED)
374         {
375             if($action == 'add')
376             {
377                 $url = $this->getReferrer($context);
378                 if ($entity instanceof KModelEntityInterface) {
379                     $url = $context->response->headers->get('Location');
380                 }
381 
382                 $context->response->setRedirect($url);
383             }
384             else $context->response->setStatus(KHttpResponse::NO_CONTENT);
385         }
386         else
387         {
388             $url     = $context->request->getReferrer();
389             $message = $entity->getStatusMessage() ? $entity->getStatusMessage() : ucfirst($action).' Action Failed';
390 
391             $context->response->setRedirect($url, $message, KControllerResponseInterface::FLASH_ERROR);
392         }
393 
394         return $entity;
395     }

397     /**
398      * Cancel action
399      *
400      * This function will unlock the row(s) and set the redirect to the referrer
401      *
402      * @param   KControllerContextInterface $context A command context object
403      * @return  KModelEntityInterface
404      */
405     protected function _actionCancel(KControllerContextInterface $context)
406     {
407         //Create the redirect
408         $context->response->setRedirect($this->getReferrer($context));
409 
410         $entity = $context->getSubject()->execute('read', $context);
411 
412         return $entity;
413     }

415     /**
416      * Add a lock flash message if the resource is locked
417      *
418      * @param   KControllerContext  $context A command context object
419      * @return  void
420      */
421     protected function _afterRead(KControllerContext $context)
422     {
423         $entity = $context->result;
424 
425         //Add the notice if the resource is locked
426         if($this->canEdit() && $entity->isLockable() && $entity->isLocked())
427         {
428             //Prevent a re-render of the message
429             if($context->request->getUrl() != $context->request->getReferrer())
430             {
431                 if($entity->isLockable() && $entity->isLocked())
432                 {
433                     $user    = $entity->getLocker();
434                     $date    = $this->getObject('date', array('date' => $entity->locked_on));
435                     $message = $this->getObject('translator')->translate(
436                         'Locked by {name} {date}', array('name' => $user->getName(), 'date' => $date->humanize())
437                     );
438 
439                     $context->response->addMessage($message, 'notice');
440                 }
441             }
442         }
443     }

445     /**
446      * Prevent editing a locked resource
447      *
448      * If the resource is locked a Retry-After header indicating the time at which the conflicting edits are expected
449      * to complete will be added. Clients should wait until at least this time before retrying the request.
450      *
451      * @param   KControllerContextInterface $context A controller context object
452      * @throws  KControllerExceptionResourceLocked If the resource is locked
453      * @return  void
454      */
455     protected function _beforeEdit(KControllerContextInterface $context)
456     {
457         if($this->isLocked())
458         {
459             $context->response->headers->set('Retry-After', $context->user->getSession()->getLifetime());
460             throw new KControllerExceptionResourceLocked('Resource is locked.');
461         }
462     }

464     /**
465      * Prevent deleting a locked resource
466      *
467      * If the resource is locked a Retry-After header indicating the time at which the conflicting edits are expected
468      * to complete will be added. Clients should wait until at least this time before retrying the request.
469      *
470      * @param   KControllerContextInterface $context A controller context object
471      * @throws  KControllerExceptionResourceLocked If the resource is locked
472      * @return  void
473      */
474     protected function _beforeDelete(KControllerContextInterface $context)
475     {
476         if($this->isLocked())
477         {
478             $context->response->headers->set('Retry-After', $context->user->getSession()->getLifetime());
479             throw new KControllerExceptionResourceLocked('Resource is locked');
480         }
481     }
