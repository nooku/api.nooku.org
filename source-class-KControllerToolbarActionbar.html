 11  * Action Controller Toolbar
 12  *
 13  * @author  Johan Janssens <https://github.com/johanjanssens>
 14  * @package Koowa\Library\Controller\Toolbar
 15  */
 16 abstract class KControllerToolbarActionbar extends KControllerToolbarAbstract
 17 {
 18     /**
 19      * Constructor
 20      *
 21      * @param   KObjectConfig $config Configuration options
 22      */
 23     public function __construct(KObjectConfig $config = null)
 24     {
 25         parent::__construct($config);
 26 
 27         //Add a title command
 28         $this->addTitle($config->title, $config->icon);
 29     }
 30 
 31     /**
 32      * Initializes the config for the object
 33      *
 34      * Called from {@link __construct()} as a first step of object instantiation.
 35      *
 36      * @param   KObjectConfig $config Configuration options
 37      * @return  void
 38      */
 39     protected function _initialize(KObjectConfig $config)
 40     {
 41         $config->append(array(
 42             'type'  => 'actionbar',
 43         ));
 44 
 45         parent::_initialize($config);
 46     }
 47 
 48     /**
 49      * Add a separator command
 50      *
 51      * @return  KControllerToolbarAbstract
 52      */
 53     public function addSeparator()
 54     {
 55         $this->_commands[] = new KControllerToolbarCommand('separator');
 56         return $this;
 57     }
 58 
 59     /**
 60      * Add default toolbar commands and set the toolbar title
 61      * .
 62      * @param KControllerContextInterface   $context A controller context object
 63      */
 64     protected function _afterRead(KControllerContextInterface $context)
 65     {
 66         $controller = $this->getController();
 67 
 68         if($controller->isEditable() && $controller->canApply()) {
 69             $this->addCommand('apply');
 70         }
 71 
 72         if($controller->isEditable() && $controller->canSave()) {
 73             $this->addCommand('save');
 74         }
 75 
 76         if($controller->isEditable() && $controller->canCancel()) {
 77             $this->addCommand('cancel',  array('attribs' => array('data-novalidate' => 'novalidate')));
 78         }
 79     }
 80 
 81     /**
 82      * Add default action commands
 83      * .
 84      * @param KControllerContextInterface $context A command context object
 85      */
 86     protected function _afterBrowse(KControllerContextInterface $context)
 87     {
 88         $controller = $this->getController();
 89 
 90         if($controller->canAdd()) {
 91             $this->addCommand('new');
 92         }
 93 
 94         if($controller->canDelete()) {
 95             $this->addCommand('delete');
 96         }
 97     }
 98 
 99     /**
100      * New toolbar command
101      *
102      * @param   KControllerToolbarCommand $command  A ControllerToolbarCommand object
103      * @return  void
104      */
105     protected function _commandNew(KControllerToolbarCommand $command)
106     {
107         if (empty($command->href))
108         {
109             $identifier    = $this->getController()->getIdentifier();
110             $command->href = 'component='.$identifier->package.'&view='.$identifier->name;
111         }
112     }
113 
114     /**
115      * Delete toolbar command
116      *
117      * @param   KControllerToolbarCommand $command  A KControllerToolbarCommand object
118      * @return  void
119      */
120     protected function _commandDelete(KControllerToolbarCommand $command)
121     {
122         $translator = $this->getObject('translator');
123         $command->append(array(
124             'attribs' => array(
125                 'data-action' => 'delete',
126                 'data-prompt' => $translator->translate('Deleted items will be lost forever. Would you like to continue?')
127             )
128         ));
129     }
130 
131     /**
132      * Enable toolbar command
133      *
134      * @param   KControllerToolbarCommand $command  A KControllerToolbarCommand object
135      * @return  void
136      */
137     protected function _commandEnable(KControllerToolbarCommand $command)
138     {
139         $command->icon = 'icon-32-publish';
140 
141         $command->append(array(
142             'attribs' => array(
143                 'data-action' => 'edit',
144                 'data-data'   => '{"enabled":1}'
145             )
146         ));
147     }
148 
149     /**
150      * Disable toolbar command
151      *
152      * @param   KControllerToolbarCommand $command  A KControllerToolbarCommand object
153      * @return  void
154      */
155     protected function _commandDisable(KControllerToolbarCommand $command)
156     {
157         $command->icon = 'icon-32-unpublish';
158 
159         $command->append(array(
160             'attribs' => array(
161                 'data-action' => 'edit',
162                 'data-data'   => '{"enabled":0}'
163             )
164         ));
165     }
166 
167     /**
168      * Export Toolbar Command
169      *
170      * @param   KControllerToolbarCommand $command  A KControllerToolbarCommand object
171      * @return  void
172      */
173     protected function _commandExport(KControllerToolbarCommand $command)
174     {
175         //Get the states
176         $states = $this->getController()->getModel()->getState()->getValues();
177 
178         unset($states['limit']);
179         unset($states['offset']);
180 
181         $states['format'] = 'csv';
182 
183         //Get the query options
184         $query  = http_build_query($states, '', '&');
185         $option = $this->getIdentifier()->package;
186         $view   = $this->getController()->getView()->getName();
187 
188         $command->href = 'option=com_'.$option.'&view='.$view.'&'.$query;
189     }
190 
191     /**
192      * Modal toolbar command
193      *
194      * @param   KControllerToolbarCommand $command  A KControllerToolbarCommand object
195      * @return  void
196      */
197     protected function _commandDialog(KControllerToolbarCommand $command)
198     {
199         $command->append(array(
200             'href'    => ''
201         ))->append(array(
202                 'attribs' => array(
203                     'class' => array('koowa-modal'),
204                     'href'  => $command->href,
205                     'data-koowa-modal'   => array('type' => 'iframe')
206                 )
207             ));
208 
209         $command->attribs['data-koowa-modal'] = json_encode($command->attribs['data-koowa-modal']);
210     }
211 }
