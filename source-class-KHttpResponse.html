203     {
204         if (!is_numeric($code) || !isset(self::$status_messages[$code]))

206             $code = is_scalar($code) ? $code : gettype($code);

532     /**
533      * Returns true if the response is worth caching under any circumstance.
534      *
535      * Responses with that are stale (Expired) or without cache validation (Last-Modified, ETag) heades are
536      * considered uncacheable.
537      *
538      * @link http://tools.ietf.org/html/rfc2616#section-14.9.1
539      * @return Boolean true if the response is worth caching, false otherwise
540      */
541     public function isCacheable()
542     {
543         if (!in_array($this->_status_code, array(200, 203, 300, 301, 302, 404, 410))) {
544             return false;
545         }
546 
547         $cache_control = (array) $this->_headers->get('Cache-Control', null, false);
548         if (isset($cache_control['no-store']) || isset($cache_control['no-cache'])) {
549             return false;
550         }
551 
552         return $this->isValidateable() || !$this->isStale();
553     }

555     /**
556      * Returns true if the response is "stale".
557      *
558      * When the responses is stale, the response may not be served from cache without first re-validating with
559      * the origin.
560      *
561      * @return Boolean true if the response is stale, false otherwise
562      */
563     public function isStale()
564     {
565         $result = true;
566         if ($maxAge = $this->getMaxAge()) {
567             $result = ($maxAge - $this->getAge()) <= 0;
568         }
569 
570         return $result;
571     }
