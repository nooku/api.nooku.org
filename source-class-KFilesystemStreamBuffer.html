 10 /**
 11  * Buffer FileSystem Stream
 12  *
 13  * The buffer:// stream  is a read-write stream that allow temporary data to be stored in a file-like.
 14  * Two types of buffers are supported :
 15  *
 16  * - buffer://temp
 17  *
 18  * A temporary file buffer that will opens a unique temp file. The location of this temporary file is determined in the
 19  * same way as the sys_get_temp_dir() function. If the stream is closed the file is removed.
 20  *
 21  * - buffer://memory[/path]
 22  *
 23  * A memory buffer that is capable of loading data from the path into the buffer, if a path is specified. If a path is
 24  * specified and the buffer is opened in writing mode, the data in memory will be flushed back into the file when the
 25  * stream is closed.
 26  *
 27  * @author  Johan Janssens <https://github.com/johanjanssens>
 28  * @package Koowa\Library\Filesystem\Stream
 29  */
 30 class KFilesystemStreamBuffer extends KFilesystemStreamAbstract

 60     /**
 61      * Opens the stream
 62      *
 63      * @throws RuntimeException If the stream cannot be opened.
 64      * @return boolean TRUE on success or FALSE on failure.
 65      */
 66     public function open()
 67     {
 68         //Create a memory buffer
 69         if($this->getType() == 'memory')
 70         {
 71             $this->_resource = '';
 72             if($path = $this->getPath())
 73             {
 74                 try {
 75                     $this->_resource = @file_get_contents($path);
 76                 } catch (Exception $e) {
 77                     $this->_resource = false;
 78                 }
 79 
 80                 if ($this->_resource === false) {
 81                     throw new RuntimeException(sprintf('Cannot get content from file : "%s"', $path));
 82                 }
 83             }
 84 
 85             $this->_position     = 0;
 86             $this->_synchronised = true;
 87 
 88             //Truncate the file to zero length
 89             if(in_array($this->getMode(), array('w', 'w+'))) {
 90                 $this->truncate(0);
 91             }
 92 
 93             //Place the file pointer at the end of the file
 94             if(in_array($this->getMode(), array('a', 'a+'))) {
 95                 $this->_position = mb_strlen($this->_resource, '8bit');
 96             }
 97 
 98             return true;
 99         }
100 
101         return parent::open();
102     }

104     /**
105      * Reads the specified number of bytes from the current position
106      *
107      * If the current position is the end-of-file, you must return an empty string.
108      *
109      * @param integer|null $bytes How many bytes of data from the current position should be returned. If NULL use the
110      *                            chunk size, default 8192
111      * @throws BadMethodCallException If read is not supported.
112      * @throws LogicException If read is not allowed.
113      * @return string If there are less than count bytes available, return as many as are available. If no more data is
114      *                available, return either FALSE or an empty string.
115      */
116     public function read($bytes = null)
117     {
118         $result = false;
119 
120         if ($this->isReadable())
121         {
122             if(!is_resource($this->_resource))
123             {
124                 $result = substr($this->_resource, $this->_position, is_int($bytes) ? $bytes : $this->getChunkSize());
125                 $this->_position += mb_strlen($result, '8bit');
126             }
127             else $result = parent::read($bytes);
128         }
129         else throw new LogicException('The stream does not allow read.');
130 
131         return $result;
132     }

134     /**
135      * Seeks to specific location in a stream.
136      *
137      * The read/write position of the stream should be updated according to the offset and whence.
138      *
139      * $whence can one of:
140      *
141      *  SEEK_SET - Set position equal to offset bytes.
142      *  SEEK_CUR - Set position to current location plus offset.
143      *  SEEK_END - Set position to end-of-file plus offset.
144      *
145      * @param integer $offset The stream offset to seek to.
146      * @param integer $whence
147      * @return boolean TRUE on success or FALSE on failure.
148      */
149     public function seek($offset, $whence = SEEK_SET)
150     {
151         if($this->getType() == 'memory')
152         {
153             switch ($whence)
154             {
155                 case SEEK_SET:
156 
157                     $this->_position = $offset;
158                     return true;
159                     break;
160 
161                 case SEEK_CUR:
162 
163                     $this->_position += $offset;
164                     return true;
165                     break;
166 
167                 case SEEK_END:
168 
169                     $this->_position = $this->_length + $offset;
170                     return true;
171                     break;
172             }
173         }
174 
175         return parent::seek($offset, $whence);
176     }

194     /**
195      *  Write to stream.
196      *
197      * If there is not enough room in the underlying stream, store as much as possible.
198      *
199      * Note : Don't forget to update the current position of the stream by number of bytes that were successfully written.
200      *
201      * @param string $data Should be stored into the underlying stream.
202      * @throws LogicException If write is not allowed.
203      * @return int Should return the number of bytes that were successfully stored, or 0 if none could be stored.
204      */
205     public function write($data)
206     {
207         $result = 0;
208 
209         if($this->getType() == 'memory')
210         {
211             if ($this->isText() && defined('PHP_WINDOWS_VERSION_MAJOR')) {
212                 $data = preg_replace('/(?<!\r)\n/', "\r\n", $data);
213             }
214 
215             $left  = substr($this->_resource, 0, $this->_position);
216             $right = substr($this->_resource, $this->_position +  mb_strlen($data, '8bit'));
217 
218             $this->_resource     = $left . $data . $right;
219             $this->_position    += mb_strlen($data, '8bit');
220             $this->_length       = mb_strlen($this->_resource, '8bit');
221             $this->_synchronised = false;
222 
223             $result = mb_strlen($data, '8bit');
224         }
225         else $result = parent::write($data);
226 
227         return $result;
228     }

236     public function eof()

238         if($this->getType() == 'memory') {

241             $result = parent::eof();

247     /**
248      * Closes the stream
249      *
250      * It must free all the resources. If there is any data to flush, you should do so
251      *
252      * @return bool Returns TRUE on success or FALSE on failure.
253      */
254     public function close()
255     {
256         if($this->getType() == 'temp')
257         {
258             if(parent::close() == true)
259             {
260                 //Only unlink temporary files
261                 if($this->getType() == 'temp') {
262                     $this->unlink();
263                 }
264 
265                 return true;
266             }
267         }
268 
269         if($this->getType() == 'memory') {
270             return $this->flush();
271         }
272 
273         return false;
274     }

276     /**
277      * Flush the data from the stream to another stream
278      *
279      * If no target stream is being passed and you have cached data that is not yet stored into the underlying storage,
280      * you should do so now
281      *
282      * @param resource|null $stream The stream resource to flush the data too
283      * @param int           $length  The total bytes to flush, if -1 the stream will be flushed until eof. The limit should
284      *                         lie within the total size of the stream.
285      * @return boolean Should return TRUE if the cached data was successfully stored (or if there was no data to store),
286      *                 or FALSE if the data could not be stored.
287      */
288     public function flush($stream = null, $length = -1)
289     {
290         $result = false;
291 
292         if($this->getType() == 'memory')
293         {
294             if(!is_resource($stream))
295             {
296                 if($path = $this->getPath() && $this->isWritable() && !$this->_synchronised)
297                 {
298                     try {
299                         $handle = @fopen($path, 'w');
300                     } catch (Exception $e) {
301                         $handle = false;
302                     }
303 
304                     if ($handle === false)
305                     {
306                         throw new RuntimeException(sprintf(
307                             'File "%s" cannot be opened with mode "%s"', $path, $this->getMode()
308                         ));
309                     }
310 
311                     //Flush the content to the underlying storage
312                     $result = fwrite($handle, $this->_resource);
313                 }
314             }
315             //Flush the content to another stream
316             else $result = fwrite($stream, $this->_resource);
317         }
318 
319         return $result;
320     }

328     public function truncate($size)
329     {
330         if($this->isWritable())
331         {
332             if($this->getType() == 'memory')
333             {
334                 if ($this->_length > $size) {

337                     $this->_resource = str_pad($this->_resource, $size, "\0", STR_PAD_RIGHT);
338                 }
339             }
340             else return parent::truncate($size);
341 
342             return true;
343         }
344 
345         return false;
